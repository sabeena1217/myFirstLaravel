git remote add origin "name of remote repo(ssh link of the repo)"

	eg:
	git remote add origin git@github.com:sabeena1217/testRepo.git
		// then the local git knows where the origin is

	origin - name of the remote local repo

git push origin master	//to upload teh master branch

after every change, add it, commit it and push it to the master using,
	git push origin master
